        // --- GAME STATE ---

        let currentIndex = 0;

        let score = 0;

        let currentAvailable = [];

        let currentAnswer = [];

 

        // --- FUNCTIONS ---

        function showScreen(id) {

            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active-screen'));

            document.getElementById(id).classList.add('active-screen');

        }

 

        function startGame() {

            document.getElementById('scoreBoard').style.display = 'block';

            loadQuestion();

        }

 

        function updateScore() {

            document.getElementById('scoreBoard').innerText = `Score: ${score}`;

        }

 

        function loadQuestion() {

            if (currentIndex >= gameData.length) {

                endGame();

                return;

            }

 

            const q = gameData[currentIndex];

 

            if (q.type === "mcq") {

                showScreen('screen-mcq');

                document.getElementById('mcq-title').innerText = q.level;

                document.getElementById('mcq-icon').innerText = q.icon;

                document.getElementById('mcq-question').innerText = q.q;

               

                const optsDiv = document.getElementById('mcq-options');

                optsDiv.innerHTML = '';

                q.options.forEach((opt, index) => {

                    const btn = document.createElement('button');

                    btn.className = 'btn btn-option';

                    btn.innerText = opt;

                    btn.onclick = () => checkMCQ(index, btn);

                    optsDiv.appendChild(btn);

                });

 

                document.getElementById('mcq-feedback').innerText = '';

                document.getElementById('mcq-next-btn').style.display = 'none';

 

            } else if (q.type === "build") {

                showScreen('screen-builder');

                document.getElementById('bld-title').innerText = q.level;

                document.getElementById('bld-icon').innerText = q.icon;

                document.getElementById('bld-instructions').innerText = q.inst;

               

                // Initialize arrays

                currentAvailable = [...q.chunks];

                currentAnswer = [];

               

                document.getElementById('bld-feedback').innerText = '';

                document.getElementById('bld-next-btn').style.display = 'none';

                document.getElementById('bld-check-btn').style.display = 'inline-block';

                document.getElementById('bld-reset-btn').style.display = 'inline-block';

 

                renderBuilder();

            }

        }

 

        // --- MCQ LOGIC ---

        function checkMCQ(selectedIndex, btnElement) {

            const q = gameData[currentIndex];

            const buttons = document.querySelectorAll('.btn-option');

            buttons.forEach(b => b.disabled = true); // Disable all

 

            const feedback = document.getElementById('mcq-feedback');

 

            if (selectedIndex === q.ans) {

                btnElement.style.backgroundColor = 'var(--dark-green)';

                btnElement.style.color = 'white';

                feedback.innerText = "🌟 Correct! Great job!";

                feedback.className = "feedback correct";

                score++;

            } else {

                btnElement.style.backgroundColor = 'var(--wrong-red)';

                btnElement.style.color = 'white';

                buttons[q.ans].style.backgroundColor = 'var(--dark-green)';

                buttons[q.ans].style.color = 'white';

                feedback.innerText = "Oops! Look at the correct answer in green.";

                feedback.className = "feedback incorrect";

            }

           

            updateScore();

            document.getElementById('mcq-next-btn').style.display = 'inline-block';

        }

 

        // --- BUILDER LOGIC ---

        function renderBuilder() {

            const ansBox = document.getElementById('bld-answer-box');

            const availBox = document.getElementById('bld-available-box');

           

            ansBox.innerHTML = '';

            availBox.innerHTML = '';

 

            currentAnswer.forEach((text, i) => {

                const chunk = document.createElement('div');

                chunk.className = 'chunk';

                chunk.innerText = text;

                chunk.onclick = () => moveChunkToAvailable(i);

                ansBox.appendChild(chunk);

            });

 

            currentAvailable.forEach((text, i) => {

                const chunk = document.createElement('div');

                chunk.className = 'chunk';

                chunk.innerText = text;

                chunk.onclick = () => moveChunkToAnswer(i);

                availBox.appendChild(chunk);

            });

        }

 

        function moveChunkToAnswer(index) {

            currentAnswer.push(currentAvailable[index]);

            currentAvailable.splice(index, 1);

            renderBuilder();

            document.getElementById('bld-feedback').innerText = ''; // clear error

        }

 

        function moveChunkToAvailable(index) {

            currentAvailable.push(currentAnswer[index]);

            currentAnswer.splice(index, 1);

            renderBuilder();

            document.getElementById('bld-feedback').innerText = '';

        }

 

        function resetBuilder() {

            const q = gameData[currentIndex];

            currentAvailable = [...q.chunks];

            currentAnswer = [];

            renderBuilder();

            document.getElementById('bld-feedback').innerText = '';

        }

 

        function checkBuilder() {

            const q = gameData[currentIndex];

            const feedback = document.getElementById('bld-feedback');

 

            if (currentAvailable.length > 0) {

                feedback.innerText = "⚠️ Please use all the pieces!";

                feedback.className = "feedback incorrect";

                return;

            }

 

            let isCorrect = true;

            for(let i = 0; i < currentAnswer.length; i++) {

                if(currentAnswer[i] !== q.correct[i]) {

                    isCorrect = false;

                    break;

                }

            }

 

            if (isCorrect) {

                feedback.innerText = "🌟 Excellent! Perfect structure!";

                feedback.className = "feedback correct";

                score++;

                updateScore();

               

                // Hide controls, show next

                document.getElementById('bld-check-btn').style.display = 'none';

                document.getElementById('bld-reset-btn').style.display = 'none';

                document.getElementById('bld-next-btn').style.display = 'inline-block';

               

                // Disable clicking on correct answers

                document.querySelectorAll('#bld-answer-box .chunk').forEach(c => c.onclick = null);

            } else {

                feedback.innerText = "❌ Not quite right. Hit Reset and try again!";

                feedback.className = "feedback incorrect";

            }

        }

 

        function nextQuestion() {

            currentIndex++;

            loadQuestion();

        }

 

        function endGame() {

            showScreen('screen-end');

            document.getElementById('scoreBoard').style.display = 'none';

            document.getElementById('final-score-text').innerText = `Your Score: ${score} / ${gameData.length}`;

           

            const msgEl = document.getElementById('final-message');

            if (score === gameData.length) {

                msgEl.innerText = "Excellent! You mastered clarity and structure! 🥇";

            } else if (score >= gameData.length * 0.7) {

                msgEl.innerText = "Good job! Keep improving your word power! 👍";

            } else {

                msgEl.innerText = "Almost there! Try again and sharpen your skills! 💪";

            }

        }
